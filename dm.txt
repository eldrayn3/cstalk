### Class
class Person:
    def __init__(self, name, age):       #constructor automatically called when creating
        self.name = name
        self.age = age
    def __str__(self):                   #alter what is printed when self is called
        return f"{self.name}, {self.age}"

personna = Person("max", 23)             #Usage
